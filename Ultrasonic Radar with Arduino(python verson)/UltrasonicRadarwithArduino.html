🔥 Arduino Code

Upload this code to your Arduino using Arduino IDE:

#include <Servo.h>

#define trigPin 8

#define echoPin 9

Servo myservo;

long duration;

int distance;

void setup() {

  Serial.begin(9600);

  myservo.attach(11);

  pinMode(trigPin, OUTPUT);

  pinMode(echoPin, INPUT);

}

void loop() {

  for (int angle = 0; angle <= 180; angle += 1) {

    myservo.write(angle);

    delay(15);

    int distance = getDistance();

    Serial.print(angle);

    Serial.print(",");

    Serial.print(distance);

    Serial.println();

  }

 

  for (int angle = 180; angle >= 0; angle -= 1) {

    myservo.write(angle);

    delay(15);

    int distance = getDistance();

    Serial.print(angle);

    Serial.print(",");

    Serial.print(distance);

    Serial.println();

  }

}

int getDistance() {

  digitalWrite(trigPin, LOW);

  delayMicroseconds(2);

  digitalWrite(trigPin, HIGH);

  delayMicroseconds(10);

  digitalWrite(trigPin, LOW);

  duration = pulseIn(echoPin, HIGH);

  distance = duration * 0.034 / 2;

  return distance;

}

✅ What it does:
Moves the servo from 0° to 180° and back while measuring distance at every angle and sending data to PC like this:

angle,distance

45,23

46,24

...

🖥️ Python (Radar Visualization) Code

Now create a new .py file in VSCode, example radar.py, and paste this:

import serial

import pygame

import math

# Set up serial communication (change COM3 to your Arduino COM port)

ser = serial.Serial('COM3', 9600, timeout=1)

# Initialize Pygame

pygame.init()

# Radar window settings

WIDTH, HEIGHT = 800, 600

screen = pygame.display.set_mode((WIDTH, HEIGHT))

pygame.display.set_caption('Ultrasonic Radar - Python Version')

# Colors

BLACK = (0, 0, 0)

GREEN = (0, 255, 0)

RED = (255, 0, 0)

# Radar settings

center_x = WIDTH // 2

center_y = HEIGHT

radius = 400

max_distance = 40  # cm

# Main loop

running = True

angle = 0

distance = 0

clock = pygame.time.Clock()

def draw_radar():

    screen.fill(BLACK)

    # Draw radar arcs

    for r in range(100, radius+1, 100):

        pygame.draw.arc(screen, GREEN, (center_x-r, center_y-r, r*2, r*2), math.pi, 2*math.pi, 1)

   

    # Draw lines at 30° intervals

    for a in range(0, 181, 30):

        x = center_x - radius * math.cos(math.radians(a))

        y = center_y - radius * math.sin(math.radians(a))

        pygame.draw.line(screen, GREEN, (center_x, center_y), (x, y), 1)

def draw_sweep(angle):

    x = center_x - radius * math.cos(math.radians(angle))

    y = center_y - radius * math.sin(math.radians(angle))

    pygame.draw.line(screen, GREEN, (center_x, center_y), (x, y), 2)

def draw_object(angle, distance):

    if distance > max_distance:

        return

    scaled_distance = (distance / max_distance) * radius

    x = center_x - scaled_distance * math.cos(math.radians(angle))

    y = center_y - scaled_distance * math.sin(math.radians(angle))

    pygame.draw.circle(screen, RED, (int(x), int(y)), 5)

while running:

    clock.tick(60)  # 60 FPS

    for event in pygame.event.get():

        if event.type == pygame.QUIT:

            running = False

    # Read Serial Data

    if ser.in_waiting:

        try:

            line = ser.readline().decode('utf-8').strip()

            if line:

                parts = line.split(',')

                if len(parts) == 2:

                    angle = int(parts[0])

                    distance = int(parts[1])

        except:

            pass

    draw_radar()

    draw_sweep(angle)

    draw_object(angle, distance)

    pygame.display.update()

ser.close()

pygame.quit()

📋 Important Notes

Change COM Port:
In the Python code line:

ser = serial.Serial('COM3', 9600, timeout=1)

  •  ➔ Change 'COM3' to whatever port your Arduino shows in Arduino IDE (e.g., COM4, COM7, etc.)
  • Match Baud Rate:
    Arduino     Serial.begin(9600); and Python Serial(..., 9600) should match.

🧪 Troubleshooting

Problem

Solution

Python throws error about serial

Check port name, baud rate. Install pyserial.

Radar window freezes

Make sure your Arduino sends constant clean serial data.

No red objects

Try placing your hand within 30–40 cm.

Servo not moving

Double-check wiring, especially 5V and GND.
