🔧 Project Title: Serial Controlled 7-Segment Display with Arduino

🧾 What You’ll Learn:

  • How to connect a 7-segment display to Arduino
  • Difference between Common Cathode and Common Anode
  • How to send numbers/characters via Serial Monitor
  • Code to display digits (0-9) and letters (A-F)
  • How to turn off the display using the “off” command

🛠 Components Required:

  • Arduino Uno/Nano
  • 7-Segment Display (Common Cathode or Common Anode)
  • 220Ω resistors (optional, one per segment)
  • Jumper Wires
  • Breadboard
  • USB cable for Arduino

Common pin:

  • For Common Cathode, connect both COM pins to GND
  • For Common Anode, connect both COM pins to 5V

🧠 How a 7-Segment Works

Each segment (a to g) is an individual LED. To light a segment:

  • Common Cathode: Give HIGH (1) to turn ON
  • Common Anode: Give LOW (0) to turn ON

📜 Arduino Code

✅ 1. Common Cathode Code

int a = 2;

int b = 3;

int c = 4;

int d = 5;

int e = 6;

int f = 7;

int g = 8;

String input;

void setup() {

  pinMode(a, OUTPUT); pinMode(b, OUTPUT); pinMode(c, OUTPUT);

  pinMode(d, OUTPUT); pinMode(e, OUTPUT); pinMode(f, OUTPUT); pinMode(g, OUTPUT);

  Serial.begin(9600);

  Serial.println("Enter a digit (0-9) or A-F. Type 'off' to clear display.");

}

void loop() {

  if (Serial.available()) {

    input = Serial.readString();

    input.trim();

    input.toUpperCase();

    Serial.print("You Entered: ");

    Serial.println(input);

    if (input == "0") {

      digitalWrite(a, HIGH); digitalWrite(b, HIGH); digitalWrite(c, HIGH);

      digitalWrite(d, HIGH); digitalWrite(e, HIGH); digitalWrite(f, HIGH);

      digitalWrite(g, LOW);

    }

    else if (input == "1") {

      digitalWrite(a, LOW); digitalWrite(b, HIGH); digitalWrite(c, HIGH);

      digitalWrite(d, LOW); digitalWrite(e, LOW); digitalWrite(f, LOW);

      digitalWrite(g, LOW);

    }

    else if (input == "2") {

      digitalWrite(a, HIGH); digitalWrite(b, HIGH); digitalWrite(c, LOW);

      digitalWrite(d, HIGH); digitalWrite(e, HIGH); digitalWrite(f, LOW);

      digitalWrite(g, HIGH);

    }

    else if (input == "3") {

      digitalWrite(a, HIGH); digitalWrite(b, HIGH); digitalWrite(c, HIGH);

      digitalWrite(d, HIGH); digitalWrite(e, LOW); digitalWrite(f, LOW);

      digitalWrite(g, HIGH);

    }

    else if (input == "4") {

      digitalWrite(a, LOW); digitalWrite(b, HIGH); digitalWrite(c, HIGH);

      digitalWrite(d, LOW); digitalWrite(e, LOW); digitalWrite(f, HIGH);

      digitalWrite(g, HIGH);

    }

    else if (input == "5") {

      digitalWrite(a, HIGH); digitalWrite(b, LOW); digitalWrite(c, HIGH);

      digitalWrite(d, HIGH); digitalWrite(e, LOW); digitalWrite(f, HIGH);

      digitalWrite(g, HIGH);

    }

    else if (input == "6") {

      digitalWrite(a, HIGH); digitalWrite(b, LOW); digitalWrite(c, HIGH);

      digitalWrite(d, HIGH); digitalWrite(e, HIGH); digitalWrite(f, HIGH);

      digitalWrite(g, HIGH);

    }

    else if (input == "7") {

      digitalWrite(a, HIGH); digitalWrite(b, HIGH); digitalWrite(c, HIGH);

      digitalWrite(d, LOW); digitalWrite(e, LOW); digitalWrite(f, LOW);

      digitalWrite(g, LOW);

    }

    else if (input == "8") {

      digitalWrite(a, HIGH); digitalWrite(b, HIGH); digitalWrite(c, HIGH);

      digitalWrite(d, HIGH); digitalWrite(e, HIGH); digitalWrite(f, HIGH);

      digitalWrite(g, HIGH);

    }

    else if (input == "9") {

      digitalWrite(a, HIGH); digitalWrite(b, HIGH); digitalWrite(c, HIGH);

      digitalWrite(d, HIGH); digitalWrite(e, LOW); digitalWrite(f, HIGH);

      digitalWrite(g, HIGH);

    }

    else if (input == "A") {

      digitalWrite(a, HIGH); digitalWrite(b, HIGH); digitalWrite(c, HIGH);

      digitalWrite(d, LOW); digitalWrite(e, HIGH); digitalWrite(f, HIGH);

      digitalWrite(g, HIGH);

    }

    else if (input == "B") {  // lowercase 'b'

      digitalWrite(a, LOW); digitalWrite(b, LOW); digitalWrite(c, HIGH);

      digitalWrite(d, HIGH); digitalWrite(e, HIGH); digitalWrite(f, HIGH);

      digitalWrite(g, HIGH);

    }

    else if (input == "C") {

      digitalWrite(a, HIGH); digitalWrite(b, LOW); digitalWrite(c, LOW);

      digitalWrite(d, HIGH); digitalWrite(e, HIGH); digitalWrite(f, HIGH);

      digitalWrite(g, LOW);

    }

    else if (input == "D") {

      digitalWrite(a, LOW); digitalWrite(b, HIGH); digitalWrite(c, HIGH);

      digitalWrite(d, HIGH); digitalWrite(e, HIGH); digitalWrite(f, LOW);

      digitalWrite(g, HIGH);

    }

    else if (input == "E") {

      digitalWrite(a, HIGH); digitalWrite(b, LOW); digitalWrite(c, LOW);

      digitalWrite(d, HIGH); digitalWrite(e, HIGH); digitalWrite(f, HIGH);

      digitalWrite(g, HIGH);

    }

    else if (input == "F") {

      digitalWrite(a, HIGH); digitalWrite(b, LOW); digitalWrite(c, LOW);

      digitalWrite(d, LOW); digitalWrite(e, HIGH); digitalWrite(f, HIGH);

      digitalWrite(g, HIGH);

    }

    else if (input == "OFF") {

      digitalWrite(a, LOW); digitalWrite(b, LOW); digitalWrite(c, LOW);

      digitalWrite(d, LOW); digitalWrite(e, LOW); digitalWrite(f, LOW);

      digitalWrite(g, LOW);

    }

    else {

      Serial.println("Invalid input! Type 0-9, A-F or 'off'.");

    }

  }

}

🔄 2. Common Anode Code

Just invert the HIGH and LOW in the above code. For example, in 0, turn ON using LOW instead of HIGH.

You can define a macro at the top to help students learn:

#define ON LOW

#define OFF HIGH

Then use it like:

digitalWrite(a, ON);

💡 How to Test

  1. Upload the code to your Arduino
  2. Open Serial Monitor (set baud rate to 9600)
  3. Type a number (0-9) or letter (A-F) and press Enter
  4. Observe the display showing your input
  5. Type off to clear the display
